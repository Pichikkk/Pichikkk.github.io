// Функция изменения фамилии
function changeLastName() {
    const newName = document.getElementById('newLastName').value;
    if (newName) {
        document.getElementById('resumeName').textContent = `Валерия ${newName}`;
    }
}

// Функция смены случайного цвета
function changeRandomColor() {
    const colors = ['#f0f8ff', '#fff0f5', '#f0fff0', '#fff8dc', '#f5f5f5'];
    const randomColor = colors[Math.floor(Math.random() * colors.length)];
    document.getElementById('resume').style.backgroundColor = randomColor;
}
